def initialize(ip:, user:, password:)
  @device_config = {
    ip: ip,
    user: user,
    password: password
  }
  @roku_ip_address = ip
  @dev_username = user
  @dev_password = password
  @url = "http://#{@roku_ip_address}"
  init()
end
